创建项目()
{
    jq检查数据合法性
    jq发参数给jsp
    jsp接收
    调数据库
    发送数据库命令{
        往projectbase中加记录
        往个人动态添加记录
        如果 这个项目属于某个实验室{
            往添加实验室动态
        }
    }
    接收结果
    jq判断是否成功
    jq返回结果到html
}

查询所有项目()
{
    jsp向数据库查询{
		遍历projectbase
    }
    统计标签个数        
    [issues 2, 4]
}
搜索()
{
    jq发关键字给search.jsp
    接收
    向数据库查询该关键字
    返回结果
    [issues 2, 4]
}

加入项目()
{
	jq发送用户名、项目id、备注
	join.jsp
	发送数据库命令{
        往projectqueue加入记录
    }
}

管理项目()
{
	编辑项目信息()
	审核人数(){
	    如果 通过：
	        projectqueue中将本记录的status置1
	        往该用户动态添加记录
	    如果 不通过：
	        删除本记录
	}
	更新项目状态(){
	    更新projectbase中状态
	}
	[删除成员()]{
		更新projectqueue中用户的status
	}
	结束项目(){
		更新projectbase状态
		更新所有该项目中用户的动态
	}
}
